import random

        palavras = ["python", "desenvolvimento", "algoritmo", "programacao", "inteligencia", "artificial", "computador"]
        
        def escolher_palavra():
            return random.choice(palavras)
        
        def exibir_palavra(palavra, letras_corretas):
            palavra_exibida = ""
            for letra in palavra:
                if letra in letras_corretas:
                    palavra_exibida += letra + " "
                else:
                    palavra_exibida += "_ "
            return palavra_exibida.strip()
        
        def jogar_forca():
            print("Bem-vindo ao Jogo da Forca!")
            
            palavra_secreta = escolher_palavra()
            letras_corretas = []
            letras_erradas = []
            tentativas = 6
            
            while tentativas > 0:
                print(f"\nVocê tem {tentativas} tentativas restantes.")
                print(f"Letras erradas: {', '.join(letras_erradas) if letras_erradas else 'Nenhuma'}")
                print("Palavra atual:", exibir_palavra(palavra_secreta, letras_corretas))
                
                letra = input("Digite uma letra: ").lower()
                
                if len(letra) != 1 or not letra.isalpha():
                    print("Por favor, digite apenas uma letra.")
                    continue
                
                if letra in letras_corretas or letra in letras_erradas:
                    print(f"Você já tentou a letra '{letra}'. Tente outra.")
                    continue
                
                if letra in palavra_secreta:
                    letras_corretas.append(letra)
                    print(f"Boa! A letra '{letra}' está na palavra.")
                else:
                    letras_erradas.append(letra)
                    tentativas -= 1
                    print(f"A letra '{letra}' não está na palavra.")
                
                if all(letra in letras_corretas for letra in palavra_secreta):
                    print(f"\nParabéns! Você adivinhou a palavra '{palavra_secreta}'. Você venceu!")
                    break
            else:
                print(f"\nVocê perdeu! A palavra era '{palavra_secreta}'.")
        
        def jogar_novamente():
            while True:
                resposta = input("\nQuer jogar novamente? (s/n): ").lower()
                if resposta == "s":
                    jogar_forca()
                elif resposta == "n":
                    print("Obrigado por jogar! Até a próxima.")
                    break
                else:
                    print("Por favor, digite 's' para sim ou 'n' para não.")
        
        jogar_forca()
        jogar_novamente()
